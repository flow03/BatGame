import json

# Відкриваємо та читаємо JSON-файл
with open('repliky.json', 'r', encoding='utf-8') as file:
    data = json.load(file)

# Отримуємо репліки з категорії "greetings"
greetings = data.get("greetings", [])

# Виводимо всі репліки з категорії "greetings"
